/**
 * –•—É–∫ –¥–ª—è —Ä–æ–±–æ—Ç–∏ –∑ localStorage –∑ TypeScript –ø—ñ–¥—Ç—Ä–∏–º–∫–æ—é
 * @template T - –¢–∏–ø –¥–∞–Ω–∏—Ö, —â–æ –∑–±–µ—Ä—ñ–≥–∞—é—Ç—å—Å—è
 * @param {string} key - –ö–ª—é—á localStorage
 * @param {T} initialValue - –ü–æ—á–∞—Ç–∫–æ–≤–µ –∑–Ω–∞—á–µ–Ω–Ω—è
 * @returns {[T, (value: T) => void]} Tuple –∑ –∑–Ω–∞—á–µ–Ω–Ω—è–º —Ç–∞ setter —Ñ—É–Ω–∫—Ü—ñ—î—é
 * @example
 * const [user, setUser] = useLocalStorage<User>('user', { id: 1, name: 'John' });
 */
function useLocalStorage<T>(
  key: string,
  initialValue: T
): [T, (value: T) => void] {
  // –û—Ç—Ä–∏–º–∞–Ω–Ω—è –∑–±–µ—Ä–µ–∂–µ–Ω–æ–≥–æ –∑–Ω–∞—á–µ–Ω–Ω—è
  const [storedValue, setStoredValue] = useState<T>(() => {
    try {
      const item = window.localStorage.getItem(key);
      return item ? JSON.parse(item) : initialValue;
    } catch (error) {
      console.error(`Error reading localStorage key "${key}":`, error);
      return initialValue;
    }
  });

  // –ó–±–µ—Ä–µ–∂–µ–Ω–Ω—è –∑–Ω–∞—á–µ–Ω–Ω—è
  const setValue = (value: T) => {
    try {
      setStoredValue(value);
      window.localStorage.setItem(key, JSON.stringify(value));
    } catch (error) {
      console.error(`Error setting localStorage key "${key}":`, error);
    }
  };

  return [storedValue, setValue];
}
